{# Shared macro for displaying a time offset (in seconds) as a human-readable "+X.Y minutes" / "+X.Y hours" / "-X.Y minutes" string. Use this whenever you need to render a signed time offset (delay, extra time, relative delay, etc.) so the formatting logic lives in one place. #} {% macro time_offset(seconds, zero_label="-", css_class="has-text-weight-semibold") %} {# Render a time offset in seconds as a human-readable string. seconds (float): the offset in seconds. Positive values are shown with "+", negative values with "-", zero shows zero_label. zero_label (str): text to display when seconds == 0 (default: "-"). css_class (str): CSS class(es) for the value span (default: Bulma semibold). #} {% if seconds > 0 %} {% set minutes = seconds / 60 %} {% if minutes < 60 %} {{ "+%.1f"|format(minutes) }} minutes {% else %} {{ "+%.1f"|format(minutes / 60) }} hours {% endif %} {% elif seconds < 0 %} {% set minutes = (-seconds) / 60 %} {% if minutes < 60 %} {{ "-%.1f"|format(minutes) }} minutes {% else %} {{ "-%.1f"|format(minutes / 60) }} hours {% endif %} {% else %} {{ zero_label }} {% endif %} {% endmacro %}
